# 可以将 16 进制串转化为与之对应的数据对象。
# 测试代码  :
# 	hb = h2bin(''' 90 90 90 90 90 90 90 90 90 ''')
def h2bin(x):
    return x.replace(' ', '').replace('\n', '').decode('hex')
